<script>
  let time = 900;
  let timerInterval;

  function startTimer() {
    clearInterval(timerInterval);
    time = 900;
    timerInterval = setInterval(() => {
      let minutes = Math.floor(time / 60);
      let seconds = time % 60;
      document.getElementById("timer").textContent =
        `${minutes}:${seconds < 10 ? "0" : ""}${seconds}`;
      time--;
      if (time < 0) clearInterval(timerInterval);
    }, 1000);
  }

  function completeSession() {
    const today = new Date().toDateString();
    const lastDate = localStorage.getItem("riseLastDate");
    let streak = parseInt(localStorage.getItem("riseStreak")) || 0;

    if (lastDate !== today) {
      streak++;
      localStorage.setItem("riseStreak", streak);
      localStorage.setItem("riseLastDate", today);
    }

    document.getElementById("streak").textContent =
      `ðŸ”¥ Current RISE Streak: ${streak} day(s)`;
    alert("Great work. Memory strengthened.");
  }

  document.getElementById("streak").textContent =
    `ðŸ”¥ Current RISE Streak: ${localStorage.getItem("riseStreak") || 0} day(s)`;
</script>
